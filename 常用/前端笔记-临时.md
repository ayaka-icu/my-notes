dateFormat() {
  const currentDate = new Date();
  currentDate.getFullYear().toString();
  return currentDate.getFullYear().toString().substring(2)
        + (currentDate.getMonth() + 1).toString().padStart(2, '0')
        + currentDate.getDate().toString().padStart(2, '0');

#### Element UI组件

##### tree树案例

```html
<el-tree
  :data="munsData"
  ref="tree"
  show-checkbox
  node-key="id"
  :default-expanded-keys="munsDataExpand"
  :default-checked-keys="munsDataChecked"
  :props="{children: 'children',label: 'label'}">
</el-tree>
```

```javascript
data() {
  return {
    result: {
        // ...
    },
    // 快捷键弹窗视图开关
    shortcutKeysDialog: false,
    // 快捷键弹窗视图数据
    shortcutViewParam: [],
    // 快捷树数据
    munsData: [],
    // 快捷树默认展开
    munsDataExpand: [],
    // 菜单树选中
    munsDataChecked: [],
    // 快捷树是否展开
    isExpand: true,
  };
},
method: {
    // 加载快捷键弹窗
    async shortcutKeysLoadingDialog() {
      this.shortcutKeysDialog = true;
      await menuTreeData().then(res => {
        this.munsData = res.data.menus
        this.munsDataChecked = res.data.checkedKeys
      })
      this.expandRetract()
    },
    // 展开/收起 tree
    expandRetract(){
      for (let i = 0; i < this.munsData.length; i++) {
        this.$refs.tree.store.nodesMap[this.munsData[i].id].expanded = this.isExpand;
      }
      this.isExpand = !this.isExpand;
    },
    // 快捷键提交修改
    shortcutKeysSubmit() {
      let checkedIds = this.$refs.tree.getCheckedKeys(true)
      /*let checkedIds = this.$refs.tree.getCheckedKeys().concat(this.$refs.tree.getHalfCheckedKeys())*/
      saveShortcutByMenuIds(checkedIds).then(res =>{
        // 更新界面数据
        this.shortcutKeysView()
      })
      this.isExpand = true
      this.shortcutKeysDialog = false;
    },
    // 快捷键弹窗取消修改
    shortcutKeysCancel() {
      this.shortcutKeysDialog = false
      this.isExpand = true
    }
}
```
